Question 2:
On ServerA, set up a Local Yum/DNF Repository using RHEL-9 ISO image mounted on the /mnt directory


# df -h (to check iso directory)
cat /etc/fstab (check if not mount)
sudo mount -o loop /path/to/rhel-9.iso /mnt

vi /etc/fstab:

/dev/loop0  /media/rheldvd xfs defaults 0 0


cd /etc/yum.repos.d
vi local.repo

[dvd-BaseOS]
name=Local Repository
baseurl=file:///mnt
enabled=1
gpgcheck=0

[dvd-AppStream]
name=Local Repository
baseurl=file:///mnt
enabled=1
gpgcheck=0
